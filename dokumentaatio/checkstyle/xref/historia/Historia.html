8   import graafinenkayttoliittyma.Laskin;
9   import java.io.File;
10  import java.io.IOException;
11  import java.util.logging.Level;
12  import java.util.logging.Logger;
13  import org.apache.commons.io.FileUtils;
14  
15  /**
16   * Luokkaan sisältyy sovelluksen ulkopuolisen tekstitiedoston kirjoittamiseen ja
17   * lukemiseen liittyvä logiikka.
18   *
19   * @author Matti
20   */
21  public class Historia {
22  
23      /**
24       * Kirjoittaa sovelluksen ulkopuoliseen tekstitiedostoon syötteen.
25       *
26       * @param history Metodille annettava syöte.
27       * @throws IOException Virhetilanne.
28       */
29      public void writeHistory(String history) throws IOException {
30          try {
31              FileUtils.writeStringToFile(new File("HISTORY.txt"), history);
32  
33          } catch (IOException ex) {
34              Logger.getLogger(Laskin.class.getName()).log(Level.SEVERE, null, ex);
35  //            IOException emptyString = new IOException("Jokin meni vikaan.");
36  //            throw emptyString;
37          }
38      }
39  
40      /**
41       * Lukee sovelluksen ulkopuolisen tekstitiedoston syötteen.
42       *
43       * @return Palauttaa tekstitiedoston sisällön String-merkkijonona.
44       */
45      public String readHistory() {
46          File test = new File("HISTORY.txt");
47          if (!test.exists()) {
48              try {
49  //                test.createNewFile(); TARPEETON, writeStringToFile luo uuden tiedoston jossei ole jo olemassa. 
50                  FileUtils.writeStringToFile(test, "HISTORY");
51              } catch (IOException ex) {
52                  Logger.getLogger(Historia.class.getName()).log(Level.SEVERE, null, ex);
53              }
54          }
55          try {
56              File file = new File("HISTORY.txt");
57              String history = FileUtils.readFileToString(file);
58              return history;
59          } catch (IOException ex) {
60              Logger.getLogger(Laskin.class.getName()).log(Level.SEVERE, null, ex);
61              return null;
62          }
63      }
64  
65  }
